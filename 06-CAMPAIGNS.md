# 9. DATA MODELS

```typescript
interface Campaign {
  id: string;
  workspaceId: string;
  
  name: string;
  description?: string;
  type: CampaignType;
  
  status: 'draft' | 'scheduled' | 'active' | 'paused' | 'completed';
  
  // Timeline
  startDate: Date;
  endDate: Date;
  launchDate?: Date;  // main event date
  
  // Config
  channels: Platform[];
  goals: ('awareness' | 'engagement' | 'conversions')[];
  
  // Phases
  phases: {
    name: string;
    startDate: Date;
    endDate: Date;
    postIds: string[];
  }[];
  
  // Metrics
  metrics?: {
    totalReach: number;
    totalEngagements: number;
    avgEngagementRate: number;
    postsPublished: number;
    postsScheduled: number;
  };
  
  createdBy: string;
  createdAt: Date;
  updatedAt: Date;
}

interface ABTest {
  id: string;
  workspaceId: string;
  
  name: string;
  platform: Platform;
  testType: 'content' | 'timing';
  
  status: 'draft' | 'running' | 'completed';
  
  variants: {
    id: string;
    name: string;
    content?: Content;
    scheduledTime?: Date;
    splitPercentage: number;
    publicationId?: string;
    
    // Results
    metrics?: {
      impressions: number;
      engagements: number;
      engagementRate: number;
    };
  }[];
  
  winnerId?: string;
  winnerSelectedAt?: Date;
  
  successMetric: 'engagement_rate' | 'clicks' | 'impressions';
  duration: number;  // hours
  
  startedAt?: Date;
  completedAt?: Date;
  
  createdAt: Date;
}

interface RecurringPost {
  id: string;
  workspaceId: string;
  
  name: string;
  status: 'active' | 'paused';
  
  // Schedule
  schedule: {
    type: 'daily' | 'weekly' | 'monthly' | 'custom';
    time: string;  // HH:MM
    timezone: string;
    daysOfWeek?: number[];  // 0-6
    dayOfMonth?: number;
    cron?: string;
  };
  
  // Content
  contentType: 'static' | 'template' | 'ai_generated' | 'rotation';
  platforms: Platform[];
  
  staticContent?: Content;
  templateId?: string;
  contentPool?: Content[];  // for rotation
  rotationIndex?: number;
  
  // Tracking
  lastRunAt?: Date;
  nextRunAt: Date;
  runCount: number;
  
  createdAt: Date;
}
```
